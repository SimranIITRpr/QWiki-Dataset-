N=int(input())
primes=[]
sprimes=[]
ans='No'
# Prime Numbers Generator
for m in range(1, N + 1):
   if m > 1:
       for n in range(2,m):
           if (m % n == 0):
               break
       else:
           primes.append(m)

# Semi-Prime Number Generator
for m in range(len(primes)):
    for n in range(m+1, len(primes)):
        if primes[m] * primes[n] <= N:
            sprimes.append(primes[n] * primes[m])
        elif primes[m] * primes[n] > N:
            break

# This part will add the semi primes to check .
sprimes.sort()
for m in range(len(sprimes)):
    for n in range(m, len(sprimes)):
        if sprimes[m] + sprimes[n] == N:
            ans='Yes'
            #print(sprimes[m],"+",sprimes[n],"=",sprimes[m]+sprimes[n])
            break
        elif sprimes[m] + sprimes[n] > N:
            break
    if(ans == 'Yes'):
        break
#print("Primes",len(primes),primes)
#print("SemiPrimes",len(sprimes),sprimes)
print(ans, end="")

   a=int((n**0.5))+1

   return False
